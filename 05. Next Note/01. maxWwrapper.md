# maxWidthWrapper for universal wrapper for fontend
```bash
npx install clsx tailwind-merge
```

components/maxWidthWrapper.tsx
```ts
import React, { ReactNode } from 'react'
import cn from '../lib/utils'


interface MaxWidthWrapperProps{
    className?:string,
    children:ReactNode

}

export const MaxWidthWrapper = ({className, children}:MaxWidthWrapperProps) => {
  return (
    <div className={cn('mx-auto w-full px-2.5 md:px-20', className)}>
        {children}
    </div>
  )
}
```
lib/utils.tsx
```ts
import clsx, { ClassValue } from "clsx"
import { twMerge } from "tailwind-merge"


const cn = (...input:ClassValue[]) => {

    return twMerge(clsx(input))

}
export default cn;
```
